    // Carousel Class
    class PremiumCarousel {
      constructor(containerId, itemWidth = 340, autoSlide = true, interval = 4000) {
        this.container = document.getElementById(containerId);
        if (!this.container) return;

        this.track = this.container.querySelector('.carousel-track');
        this.items = Array.from(this.track.children);
        if (this.items.length === 0) return;

        this.itemWidth = itemWidth;
        this.currentIndex = 0;
        this.autoSlide = autoSlide;
        this.interval = interval;
        this.autoSlideTimer = null;

        this.init();
      }

      init() {
        this.createNavigation();
        this.updatePosition();
        this.addEventListeners();
        if (this.autoSlide) this.startAutoSlide();
        this.updateIndicators();
      }

      createNavigation() {
        const prevBtn = document.createElement('button');
        prevBtn.className = 'carousel-nav carousel-prev';
        prevBtn.innerHTML = 'â€¹';
        prevBtn.setAttribute('aria-label', 'Previous');

        const nextBtn = document.createElement('button');
        nextBtn.className = 'carousel-nav carousel-next';
        nextBtn.innerHTML = 'â€º';
        nextBtn.setAttribute('aria-label', 'Next');

        const indicators = document.createElement('div');
        indicators.className = 'carousel-indicators';

        this.items.forEach((_, index) => {
          const dot = document.createElement('button');
          dot.className = 'carousel-indicator';
          dot.setAttribute('aria-label', `Go to slide ${index + 1}`);
          indicators.appendChild(dot);
        });

        this.container.appendChild(prevBtn);
        this.container.appendChild(nextBtn);
        this.container.appendChild(indicators);

        this.prevBtn = prevBtn;
        this.nextBtn = nextBtn;
        this.indicators = indicators;
      }

      addEventListeners() {
        this.prevBtn.addEventListener('click', () => this.prev());
        this.nextBtn.addEventListener('click', () => this.next());

        this.indicators.querySelectorAll('.carousel-indicator').forEach((dot, index) => {
          dot.addEventListener('click', () => this.goTo(index));
        });

        this.container.addEventListener('mouseenter', () => this.pauseAutoSlide());
        this.container.addEventListener('mouseleave', () => this.resumeAutoSlide());
      }

      updatePosition() {
        const gap = window.innerWidth > 768 ? 20 : 0;
        const itemWidth = this.items[0].getBoundingClientRect().width + gap;
        this.track.style.transform = `translateX(-${this.currentIndex * itemWidth}px)`;
      }

      updateIndicators() {
        const dots = this.indicators.querySelectorAll('.carousel-indicator');
        dots.forEach((dot, index) => {
          dot.classList.toggle('active', index === this.currentIndex);
        });
      }

      prev() {
        this.currentIndex = this.currentIndex > 0 ? this.currentIndex - 1 : this.items.length - 1;
        this.updatePosition();
        this.updateIndicators();
        this.resetAutoSlide();
      }

      next() {
        this.currentIndex = (this.currentIndex + 1) % this.items.length;
        this.updatePosition();
        this.updateIndicators();
        this.resetAutoSlide();
      }

      goTo(index) {
        this.currentIndex = index;
        this.updatePosition();
        this.updateIndicators();
        this.resetAutoSlide();
      }

      startAutoSlide() {
        this.autoSlideTimer = setInterval(() => this.next(), this.interval);
      }

      pauseAutoSlide() {
        if (this.autoSlideTimer) {
          clearInterval(this.autoSlideTimer);
          this.autoSlideTimer = null;
        }
      }

      resumeAutoSlide() {
        if (this.autoSlide && !this.autoSlideTimer) this.startAutoSlide();
      }

      resetAutoSlide() {
        if (this.autoSlide) {
          this.pauseAutoSlide();
          this.startAutoSlide();
        }
      }
    }
